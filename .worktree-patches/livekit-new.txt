import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import type { TokenSourceResponsePayload } from "livekit-client";
import { AccessToken } from "livekit-server-sdk";

import { requireAuth, handleAuthError } from "@/lib/auth/middleware";
import { prisma } from "@/lib/db/prisma";
import {
  buildChatVoiceJobMetadata,
  buildChatVoiceParticipantIdentity,
  buildChatVoiceRoomConfiguration,
  buildChatVoiceRoomName,
  getLiveKitServerConfig,
  resolveLiveKitAgentName,
} from "@/lib/livekit/chat-voice";

const querySchema = z.object({
  sessionId: z.string().min(1),
});

function jsonResponse(body: object, init?: ResponseInit) {
  return NextResponse.json(body, {
    ...init,
    headers: {
      "Cache-Control": "no-store",
      ...(init?.headers ?? {}),
    },
  });
}

export async function POST(req: NextRequest) {
  try {
    const liveKit = getLiveKitServerConfig();

    if (!liveKit) {
      return jsonResponse(
        {
          error:
            "LiveKit voice is unavailable until LIVEKIT_URL, LIVEKIT_API_KEY, and LIVEKIT_API_SECRET are configured.",
          code: "livekit_not_configured",
        },
        { status: 503 },
      );
    }

    const auth = await requireAuth();
    const { sessionId } = querySchema.parse({
      sessionId: req.nextUrl.searchParams.get("sessionId"),
    });

    const session = await prisma.chatSession.findUnique({
      where: { id: sessionId },
      include: {
        agent: {
          select: {
            id: true,
            slug: true,
            envVars: true,
          },
        },
      },
    });

    if (!session || session.userId !== auth.userId) {
      return jsonResponse({ error: "Session not found" }, { status: 404 });
    }

    const liveKitAgentName = resolveLiveKitAgentName({
      agentEnvVars: session.agent.envVars,
      fallbackAgentName: liveKit.defaultAgentName,
    });

    if (!liveKitAgentName) {
      return jsonResponse(
        {
          error:
            "LiveKit voice is unavailable until LIVEKIT_AGENT_NAME is configured or the agent envVars include livekitAgentName.",
          code: "livekit_agent_not_configured",
        },
        { status: 503 },
      );
    }

    const roomName = buildChatVoiceRoomName(session.id);
    const participantIdentity = buildChatVoiceParticipantIdentity(session.id);
    const participantName = "Randi chat user";
    const dispatchMetadata = buildChatVoiceJobMetadata({
      sessionId: session.id,
      userId: auth.userId,
      agentId: session.agent.id,
      agentSlug: session.agent.slug,
    });

    const token = new AccessToken(liveKit.apiKey, liveKit.apiSecret, {
      identity: participantIdentity,
      name: participantName,
      metadata: JSON.stringify({
        source: "agent-platform-chat",
        sessionId: session.id,
      }),
    });

    token.addGrant({ roomJoin: true, room: roomName });
    token.roomConfig = buildChatVoiceRoomConfiguration(liveKitAgentName, dispatchMetadata);

    const payload: TokenSourceResponsePayload = {
      server_url: liveKit.serverUrl,
      participant_token: await token.toJwt(),
    };

    return jsonResponse(payload);
  } catch (error) {
    if (error instanceof z.ZodError) {
      return jsonResponse({ error: "Invalid LiveKit token request" }, { status: 400 });
    }

    return handleAuthError(error);
  }

export function VoiceChatPanel({ agentName, sessionId }: VoiceChatPanelProps) {
  const [error, setError] = useState<string | null>(null);
  const [isMicrophoneEnabled, setIsMicrophoneEnabled] = useState(false);

  const tokenSource = useMemo(
    () =>
      TokenSource.endpoint(`/api/livekit/token?sessionId=${encodeURIComponent(sessionId)}`, {
        headers: {
          "X-Requested-With": "XMLHttpRequest",
        },
      }),
    [sessionId],
  );

  const session = useSession(tokenSource);

  useEffect(() => {
    return () => {
      void session.end().catch(() => undefined);
    };
  }, [session]);

  useEffect(() => {
    if (!session.isConnected) {
      setIsMicrophoneEnabled(false);
    }
  }, [session.connectionState, session.isConnected]);

  const status: VoiceStatus = useMemo(() => {
    if (error) {
      return "error";
    }

    if (session.connectionState === "connecting") {
      return "connecting";
    }

    if (session.isConnected) {
      return "live";
    }

    return "idle";
  }, [error, session.connectionState, session.isConnected]);

  const helperText = useMemo(() => {
    if (error) {
      return error;
    }

    if (status === "connecting") {
      return "Connecting to LiveKit and dispatching the configured voice agent for this chat session.";
    }

    if (status === "live") {
      return "Your voice session is connected. Speak naturally and Randi will respond over LiveKit audio.";
    }

    return `Start a LiveKit Agents voice session for ${agentName} without changing the existing text-chat flow below.`;
  }, [agentName, error, status]);

  const handleStart = useCallback(async () => {
    setError(null);

    try {
      await session.start({
        tracks: {
          microphone: {
            enabled: true,
          },
        },
      });
      setIsMicrophoneEnabled(true);
    } catch (caughtError) {
      setError(getErrorMessage(caughtError));
      setIsMicrophoneEnabled(false);
    }
  }, [session]);

  const handleDisconnect = useCallback(async () => {
    setError(null);

    try {
      await session.end();
      setIsMicrophoneEnabled(false);
    } catch (caughtError) {
      setError(getErrorMessage(caughtError));
    }
  }, [session]);

  const handleToggleMicrophone = useCallback(async () => {
    setError(null);

    try {
      const nextEnabled = !session.room.localParticipant.isMicrophoneEnabled;
      await session.room.localParticipant.setMicrophoneEnabled(nextEnabled);
      setIsMicrophoneEnabled(nextEnabled);
    } catch (caughtError) {
      setError(getErrorMessage(caughtError));
    }
  }, [session]);

  return (
    <SessionProvider session={session}>
      <div className="rounded-2xl border border-border bg-card/40 p-4">
        <div className="flex items-start justify-between gap-4 flex-wrap">
          <div>
            <p className="text-sm font-semibold text-primary">LiveKit voice</p>
            <h3 className="mt-2 text-xl font-black tracking-tight">Voice session for this chat</h3>
            <p className="mt-2 max-w-2xl text-sm leading-relaxed text-muted-foreground">
              {helperText}
            </p>
          </div>
          <span
            className={`inline-flex rounded-full border px-3 py-1 text-sm font-semibold ${STATUS_STYLES[status]}`}
          >
            {STATUS_LABELS[status]}
          </span>
        </div>

        <div className="mt-4 grid gap-4 lg:grid-cols-[minmax(0,1fr)_320px]">
          <div className="rounded-2xl border border-border/60 bg-background/40 p-4">
            <p className="text-xs font-semibold uppercase tracking-[0.22em] text-muted-foreground">
              Session controls
            </p>
            <div className="mt-3 flex flex-wrap gap-3">
              {!session.isConnected ? (
                <button
                  type="button"
                  onClick={handleStart}
                  disabled={status === "connecting"}
                  className="rounded-xl bg-primary px-4 py-2 text-sm font-semibold text-white transition hover:bg-primary/90 disabled:cursor-not-allowed disabled:opacity-60"

import { RoomAgentDispatch, RoomConfiguration } from "@livekit/protocol";

export const CHAT_VOICE_ROOM_PREFIX = "chat-voice";
export const LIVEKIT_AGENT_ENV_KEY = "livekitAgentName";

export interface LiveKitServerConfig {
  apiKey: string;
  apiSecret: string;
  defaultAgentName: string | null;
  serverUrl: string;
}

export interface ChatVoiceMetadataInput {
  agentId: string;
  agentSlug: string;
  sessionId: string;
  userId: string;
}

type LiveKitConfigEnv = Record<string, string | undefined>;

export function getLiveKitServerConfig(
  env: LiveKitConfigEnv = process.env,
): LiveKitServerConfig | null {
  const serverUrl = env.LIVEKIT_URL?.trim();
  const apiKey = env.LIVEKIT_API_KEY?.trim();
  const apiSecret = env.LIVEKIT_API_SECRET?.trim();

  if (!serverUrl || !apiKey || !apiSecret) {
    return null;
  }

  return {
    serverUrl,
    apiKey,
    apiSecret,
    defaultAgentName: env.LIVEKIT_AGENT_NAME?.trim() || null,
  };
}

export function buildChatVoiceRoomName(sessionId: string): string {
  return `${CHAT_VOICE_ROOM_PREFIX}-${sessionId}`;
}

export function buildChatVoiceParticipantIdentity(sessionId: string): string {
  return `chat-user-${sessionId}`;
}

export function parseAgentEnvVars(envVars?: string | null): Record<string, unknown> {
  if (!envVars) {
    return {};
  }

  try {
    const parsed = JSON.parse(envVars);
    return parsed && typeof parsed === "object" && !Array.isArray(parsed) ? parsed : {};
  } catch {
    return {};
  }
}

export function resolveLiveKitAgentName(options: {
  agentEnvVars?: string | null;
  fallbackAgentName?: string | null;
}): string | null {
  const parsedEnv = parseAgentEnvVars(options.agentEnvVars);
  const override = parsedEnv[LIVEKIT_AGENT_ENV_KEY];

  if (typeof override === "string" && override.trim()) {
    return override.trim();
  }

  return options.fallbackAgentName?.trim() || null;
}

export function buildChatVoiceJobMetadata(input: ChatVoiceMetadataInput): string {
  return JSON.stringify({
    source: "agent-platform-chat",
    sessionId: input.sessionId,
    userId: input.userId,
    agentId: input.agentId,
    agentSlug: input.agentSlug,
  });
}

export function buildChatVoiceRoomConfiguration(agentName: string, metadata: string) {
  return new RoomConfiguration({
    agents: [
      new RoomAgentDispatch({
        agentName,
        metadata,
      }),
    ],
  });
}
